\usepackage{import}
\usepackage{coseoul}

% See this Stack Exchange question for information on how this macro works
% http://tex.stackexchange.com/questions/245348/recursively-import-sub-document-at-arbitrary-level

\newcommand{\subimportlevel}[2]{
\edef\mycurrentlevel{\thecurrentlevel}
  \bgroup
  \subimport*{#1}{#2}
  \egroup
\setcounter{currentlevel}{\mycurrentlevel}
}
